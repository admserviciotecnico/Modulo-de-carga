  // === 1) Lectura de Excel con SheetJS ===
  function leerExcelBOM(event) {
    const file = event.target.files[0];
    if (!file) return;

    const reader = new FileReader();
    reader.onload = function(e) {
      const data = new Uint8Array(e.target.result);
      const workbook = XLSX.read(data, { type: 'array' });

      const firstSheetName = workbook.SheetNames[0];
      const worksheet = workbook.Sheets[firstSheetName];

      const jsonData = XLSX.utils.sheet_to_json(worksheet, { defval: "" });

      if (!jsonData.length) {
        alert("El Excel parece estar vacío o no se pudo leer.");
        return;
      }

      const headers = Object.keys(jsonData[0]);
      function findCol(posibles) {
        const lowerHeaders = headers.map(h => h.toLowerCase());
        for (const posible of posibles) {
          const idx = lowerHeaders.indexOf(posible.toLowerCase());
          if (idx !== -1) return headers[idx];
        }
        return null;
      }

      const colItem      = findCol(["ITEM", "ÍTEM", "POS", "Nro Item"])      || headers[0];
      const colCodigo    = findCol(["CODIGO", "CÓDIGO", "PART NUMBER"])      || headers[1];
      const colDescripcion = findCol(["DESCRIPCION", "DESCRIPCIÓN", "DESCRIPTION"]) || headers[2];
      const colCantidad  = findCol(["CANTIDAD", "QTY", "CANT"])              || headers[3];
      const colTipo      = findCol(["TIPO", "Tipo"])                         || null;
      const colMaterial  = findCol(["MATERIAL", "Mat", "MATERIALES"])        || null;
      const colPeso      = findCol(["PESO", "Weight", "Masa"])               || null;

      bomItems = jsonData.map((row, index) => {
        return {
          id: index + 1,
          item: row[colItem] || (index + 1),
          codigo: row[colCodigo] || "",
          descripcion: row[colDescripcion] || "",
          cantidad: row[colCantidad] || 1,
          tipo: colTipo ? (row[colTipo] || "") : "",
          material: colMaterial ? (row[colMaterial] || "") : "",
          peso: colPeso ? (row[colPeso] || "") : "",
          plano: "",
          procesos: []
        };
      });

      plantillaAprobada = false;
      itemSeleccionadoId = null;
      document.getElementById("estado-aprobacion").textContent = "BORRADOR";
      renderTablaBom();
      document.getElementById("detalle-item").innerHTML =
        '<p class="small">Seleccioná un ítem en la tabla para editarlo.</p>';
    };

    reader.readAsArrayBuffer(file);
  }

  // === 2) Render de tabla BOM ===
  function renderTablaBom() {
    const cont = document.getElementById("bom-container");
    if (!bomItems.length) {
      cont.innerHTML = '<p class="small">Todavía no hay BOM cargado.</p>';
      return;
    }

    let html = '<table><thead><tr>' +
      '<th>Ítem</th><th>Código parte</th><th>Descripción</th><th>Cant.</th>' +
      '<th>Tipo</th><th>Material</th><th>Peso</th>' +
      '<th>Plano DWG</th><th>Procesos</th>' +
      '</tr></thead><tbody>';

    bomItems.forEach(b => {
      const procesosCount = (b.procesos || []).length;
      html += `<tr class="selectable ${b.id === itemSeleccionadoId ? 'selected' : ''}"
                  onclick="seleccionarItem(${b.id})">
        <td>${b.item}</td>
        <td>${b.codigo}</td>
        <td>${b.descripcion}</td>
        <td style="text-align:right;">${b.cantidad}</td>
        <td>${b.tipo || ''}</td>
        <td>${b.material || ''}</td>
        <td>${b.peso || ''}</td>
        <td>${b.plano ? b.plano : '<span class="small">Sin plano</span>'}</td>
        <td>${procesosCount > 0 ? procesosCount + ' proceso(s)' : '<span class="small">Sin definir</span>'}</td>
      </tr>`;
    });

    html += '</tbody></table>';
    cont.innerHTML = html;
  }

  function renderFilaProceso(idx, tipo, nombreProceso, proveedor, disabled) {
    const tipos = Object.keys(maestroProcesos);
    let html = `<tr>
      <td>${idx + 1}</td>
      <td>
        <select onchange="onCambioTipo(${idx}, this.value)" ${disabled}>
          <option value="">(tipo)</option>`;
    tipos.forEach(t => {
      html += `<option value="${t}" ${t === tipo ? 'selected' : ''}>${t}</option>`;
    });
    html += `</select>
      </td>
      <td>
        <select onchange="onCambioProceso(${idx}, this.value)" ${disabled}>
          <option value="">(proceso)</option>`;

    if (tipo && maestroProcesos[tipo]) {
      maestroProcesos[tipo].forEach(proc => {
        html += `<option value="${proc.nombre}" ${proc.nombre === nombreProceso ? 'selected' : ''}>
                   ${proc.nombre}
                 </option>`;
      });
    }

    html += `</select>
      </td>
      <td>
        <input type="text" value="${proveedor || ''}" readonly>
      </td>
      <td>
        <button onclick="eliminarProceso(${idx})" ${disabled}>X</button>
      </td>
    </tr>`;

    return html;
  }

  // Simulación de "subir" DWG y guardarlo en la carpeta de planos
  function subirPlanoDWG() {
    const item = getItemSeleccionado();
    if (!item || plantillaAprobada) return;

    const inputDwg = document.getElementById("input-dwg");
    if (!inputDwg || !inputDwg.files || !inputDwg.files[0]) {
      alert("Seleccioná un archivo DWG primero.");
      return;
    }

    const file = inputDwg.files[0];
    const fileName = file.name; // p.ej. AB200-001.dwg
    const baseRutaPlanos = '\\\\192.168.88.223\\Ingenier\u00eda\\Producci\u00f3n\\06_Planos_para_fabricacion\\';
    const rutaDestino = baseRutaPlanos + fileName;

    // En demo solo guardamos la ruta en memoria:

    // Mostramos la ruta en el campo de texto
    const inputPlano = document.getElementById("input-plano");
    if (inputPlano) {
      inputPlano.value = rutaDestino;
    }

    document.getElementById("span-dwg-demo-nombre").textContent = fileName;

    // En sistema real: aquí iría una llamada AJAX / fetch al backend
    // enviando el archivo DWG para copiarlo en el servidor.

    alert("Plano DWG 'subido' (demo).\nRuta simulada:\n" + rutaDestino);

  function agregarProceso() {
    const item = getItemSeleccionado();
    if (!item || plantillaAprobada) return;

    item.procesos = item.procesos || [];
    if (item.procesos.length >= 10) {
      document.getElementById("msg-max-procesos").textContent =
        "Se alcanzó el máximo de 10 procesos para este ítem.";
      return;
    } else {
      document.getElementById("msg-max-procesos").textContent = "";
    }

    item.procesos.push({
      tipo: "",
      nombre: "",
      proveedor: ""
    });
    renderDetalleItem();
    renderTablaBom();
  }

  function eliminarProceso(idx) {
    const item = getItemSeleccionado();
    if (!item || plantillaAprobada) return;
    item.procesos.splice(idx, 1);
    renderDetalleItem();
    renderTablaBom();
  }

  function onCambioTipo(idx, nuevoTipo) {
    const item = getItemSeleccionado();
    if (!item || plantillaAprobada) return;
    const proc = item.procesos[idx];
    proc.tipo = nuevoTipo;
    proc.nombre = "";
    proc.proveedor = "";
    renderDetalleItem();
    renderTablaBom();
  }

  function onCambioProceso(idx, nombreProceso) {
    const item = getItemSeleccionado();
    if (!item || plantillaAprobada) return;
    const proc = item.procesos[idx];
    proc.nombre = nombreProceso;

    const lista = maestroProcesos[proc.tipo] || [];
    const encontrado = lista.find(p => p.nombre === nombreProceso);
    proc.proveedor = encontrado ? encontrado.proveedor : "";

    renderDetalleItem();
    renderTablaBom();
  }

  function aprobarPlantilla() {
    if (!bomItems.length) {
      alert("No hay BOM cargado.");
      return;
    }
    plantillaAprobada = true;
    document.getElementById("estado-aprobacion").textContent = "APROBADA";
    alert("Plantilla aprobada (demo). En el sistema real se congela la lista y se graba en la base de datos.");
    renderDetalleItem();
  }
